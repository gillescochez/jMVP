jMVP.each(['a', 'b'], function(sValue, nIdx) {
     console.log(sValue);

// Using the context parameter

     this.log = function(sValue) {
         console.log(sValue);
     }

     jMVP.each(['a', 'b'], function(sValue) {
         this.log(sValue);
     }, this);
